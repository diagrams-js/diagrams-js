## Usage Example

```typescript
import { Diagram } from "diagrams-js";
import { ComputeEngine, AppEngine, Functions, GKE } from "diagrams-js/gcp/compute";
import { CloudSQL, Bigtable } from "diagrams-js/gcp/database";
import { Storage } from "diagrams-js/gcp/storage";
import { VirtualPrivateCloud, CloudLoadBalancing } from "diagrams-js/gcp/network";

const diagram = Diagram("GCP Architecture", { direction: "TB" });

const vpc = diagram.add(VirtualPrivateCloud("VPC"));
const lb = diagram.add(CloudLoadBalancing("Load Balancer"));
const gke = diagram.add(GKE("Kubernetes Cluster"));
const app = diagram.add(AppEngine("App Engine"));
const api = diagram.add(Functions("Cloud Functions"));
const sql = diagram.add(CloudSQL("Cloud SQL"));
const nosql = diagram.add(Bigtable("Bigtable"));
const storage = diagram.add(Storage("Cloud Storage"));

lb.to([gke, app]);
gke.to(api);
api.to([sql, nosql]);
api.to(storage);

const svg = await diagram.render();
diagram.destroy();
```

:::note
For a complete list of all GCP nodes, see the [Python diagrams GCP documentation](https://diagrams.mingrammer.com/docs/nodes/gcp). All node classes listed there are available in diagrams-js with the same class names.
:::

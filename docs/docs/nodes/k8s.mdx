## Usage Example

```typescript
import { Diagram } from "diagrams-js";
import { Deployment, Pod, ReplicaSet } from "diagrams-js/k8s/compute";
import { Service, Ingress } from "diagrams-js/k8s/network";
import { PV, PVC } from "diagrams-js/k8s/storage";
import { HPA } from "diagrams-js/k8s/clusterconfig";

const diagram = Diagram("Kubernetes Architecture", { direction: "TB" });

// Ingress and Service
const ingress = diagram.add(Ingress("ingress.example.com"));
const svc = diagram.add(Service("web-svc"));

// Pods managed by ReplicaSet and Deployment
const pods = [
  diagram.add(Pod("web-pod-1")),
  diagram.add(Pod("web-pod-2")),
  diagram.add(Pod("web-pod-3")),
];

const rs = diagram.add(ReplicaSet("web-rs"));
const deploy = diagram.add(Deployment("web-deploy"));
const hpa = diagram.add(HPA("web-hpa"));

// Storage
const pvc = diagram.add(PVC("web-pvc"));
const pv = diagram.add(PV("web-pv"));

// Connections
ingress.to(svc);
svc.to(pods);
pods.forEach((pod) => pod.from(rs));
rs.from(deploy);
deploy.from(hpa);
pods.forEach((pod) => pod.with(pvc));
pvc.from(pv);

const svg = await diagram.render();
diagram.destroy();
```

:::note
For a complete list of all Kubernetes nodes, see the [Python diagrams K8s documentation](https://diagrams.mingrammer.com/docs/nodes/k8s). All node classes listed there are available in diagrams-js with the same class names.
:::

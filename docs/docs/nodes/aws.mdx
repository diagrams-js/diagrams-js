## Usage Example

```typescript
import { Diagram } from "diagrams-js";
import { EC2, Lambda, ECS } from "diagrams-js/aws/compute";
import { RDS, Dynamodb } from "diagrams-js/aws/database";
import { S3 } from "diagrams-js/aws/storage";
import { VPC, ALB } from "diagrams-js/aws/network";

const diagram = Diagram("AWS Architecture", { direction: "TB" });

const vpc = diagram.add(VPC("VPC"));
const alb = diagram.add(ALB("ALB"));
const web = diagram.add(EC2("Web Server"));
const api = diagram.add(Lambda("API"));
const db = diagram.add(RDS("Database"));
const cache = diagram.add(Dynamodb("Cache"));
const storage = diagram.add(S3("Storage"));

alb.to(web);
web.to(api);
api.to([db, cache]);
api.to(storage);

const svg = await diagram.render();
diagram.destroy();
```

:::note
For a complete list of all AWS nodes, see the [Python diagrams AWS documentation](https://diagrams.mingrammer.com/docs/nodes/aws). All node classes listed there are available in diagrams-js with the same class names.
:::
